def pickdog(want_type):

    for i in range(len(dog_act)):
        if want_type in dog_act[i]:
            filtered_list.append(dog_breeds[i])
            if filtered_list==[]:
                print("Sorry, no dogs quite fit that recommendation, either use more broad terms, or synonyms")#This just means none of our dogs specifically do this
            print("These are the dogs that fit your requests, here are their pictures")
            print(dog_breeds[i])
            open_image(dog_image[i])
